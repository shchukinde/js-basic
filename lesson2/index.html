Префиксная форма инкремента\декремента возвращает новое значение, в то время как постфиксная форма возвращает старое
var a = 1, b = 1, c, d;
c = ++a; alert(c); // 2 , т.к. префиксный инкремент, "c" присваивается новое значение "a", т.е. 1+1 = 2
d = b++; alert(d); // 1 , т.к. постфиксный инкремент, "d" присваивается старое значение "b", т.е. 1, но b при этом стало 1+1 = 2
c = (2+ ++a); alert(c); // 5, т.к. т.к. префиксный инкремент "a", т.е. a = 2+1 = 3, c = 2 + 3 = 5
d = (2+ b++); alert(d); // 4, т.к. постфиксный инкремент "b", который возвращает 2 и получается d = 2 + 2 = 4, но при этом b = 2 + 1
alert(a); // 3
alert(b); // 3
